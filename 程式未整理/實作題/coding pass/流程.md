```cpp linenos title:"解答"
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h> 
#include <iostream>

using namespace std;


int main() {
	int len, tmp, num = 0;
	bool dec = true, oct = true, bin = true;
	// 0 < en <32

	while (1) {
		cin >> len;

		if (len == 0) break;

		for (int i = 0; i < len; i++) {
			cin >> tmp;
			num = num * 2 + tmp;
			if ((len - i - 1) % 4 == 0) {
				if (num > 2) bin = false;
				if (num > 7) oct = false;
				if (num > 9) dec = false;
				num = 0;
			}
		}

		cout << 'H';
		if (dec) cout << 'D';
		if (oct) cout << 'O';
		if (bin) cout << 'B';
		cout << endl;
	}
	
	return 0;
}
```
